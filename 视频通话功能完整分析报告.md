### 用户端流程

#### 4. 用户接收视频邀请

**页面**：`/pages/support/index`（旧版，同时支持客服和用户）

```javascript
function connectSocket() {
  closeSocket()
  try {
    const conversationId = store.state.supportConversationId
    socketTask.value = connectChatSocket('support', conversationId, (payload) => {
      // ✅ 检查是否是视频邀请
      if (isVideoInvite(payload)) {
        handleVideoInvite(payload)
        return
      }
      
      // 处理普通消息...
    }, {
      onOpen: () => { socketStatus.value = 'online' },
      onClose: () => { socketStatus.value = 'offline' },
      onError: () => { socketStatus.value = 'offline' }
    })
  } catch (error) {
    console.error('WebSocket连接失败:', error)
  }
}

function isVideoInvite(payload: ChatRealtimePayload): payload is VideoInviteEvent {
  return 'eventType' in payload && payload.eventType === 'video_invite'
}

function handleVideoInvite(invite: VideoInviteEvent) {
  // ✅ 检查是否是当前会话
  if (invite.channelId !== store.state.supportConversationId) return
  // ✅ 防止重复处理
  if (handledVideoInvites.has(invite.sessionId)) return
  handledVideoInvites.add(invite.sessionId)

  // ✅ 检查是否是发起者本人（避免自己接自己的电话）
  const currentUsername = store.state.currentUser?.username
  if (!currentUsername || invite.initiatorUsername === currentUsername) return

  // ✅ 弹出接听/拒绝对话框
  uni.showModal({
    title: '视频通话',
    content: `${invite.initiatorUsername} 正在呼叫您...`,
    confirmText: '接听',
    cancelText: '拒绝',
    async success(result) {
      if (!result.confirm) return
      try {
        // ✅ 获取视频会话引导信息
        const bootstrap = await fetchVideoSessionBootstrap(invite.sessionId)
        const encoded = encodeURIComponent(JSON.stringify(bootstrap))
        // ✅ 跳转到视频通话页面
        uni.navigateTo({ url: `/pages/video-call/index?bootstrap=${encoded}` })
      } catch (error) {
        uni.showToast({
          title: error instanceof Error ? error.message : '无法加入通话',
          icon: 'none'
        })
      }
    }
  })
}
```

```javascript
// ✅ 加入TRTC房间
async function joinRoom() {
  const client = TRTC.create()
  trtc.value = client
  
  // ✅ 进入房间
  await client.enterRoom({
    sdkAppId: Number(bootstrap.sdkAppId),
    userId: bootstrap.userId,
    userSig: bootstrap.userSig,
    strRoomId: bootstrap.session.roomId,  // ✅ 双方使用相同的roomId
    scene: TRTC.TYPE.SCENE_RTC,
    autoReceiveAudio: true,
    autoReceiveVideo: false
  })
  
  // ✅ 启动本地设备
  await startLocalDevices(client)
}

async function startLocalDevices(client: any) {
  // ✅ 启动本地摄像头
  try {
    await client.startLocalVideo({ view: 'local-video-view' })
    localVideoStarted.value = true
  } catch (error) {
    mutedVideo.value = true
  }
  
  // ✅ 启动本地麦克风
  try {
    await client.startLocalAudio()
  } catch (error) {
    mutedAudio.value = true
  }
}

// ✅ 监听远端用户
function bindTrtcEvents(client: any) {
  client.on(TRTC.EVENT.REMOTE_VIDEO_AVAILABLE, (event: any) => {
    // ✅ 远端用户开启摄像头
    hasRemoteVideo.value = true
    client.startRemoteVideo({
      userId: event.userId,
      streamType: event.streamType,
      view: 'remote-video-view'  // ✅ 在远端视频区域显示
    })
  })
  
  client.on(TRTC.EVENT.REMOTE_VIDEO_UNAVAILABLE, (event: any) => {
    // ✅ 远端用户关闭摄像头
    hasRemoteVideo.value = false
    client.stopRemoteVideo({ userId: event.userId, streamType: event.streamType })
  })
}

// ✅ 控制按钮
async function toggleAudio() {
  if (mutedAudio.value) {
    await trtc.value.startLocalAudio()
    mutedAudio.value = false
  } else {
    await trtc.value.stopLocalAudio()
    mutedAudio.value = true
  }
}

async function toggleVideo() {
  if (mutedVideo.value) {
    await trtc.value.startLocalVideo({ view: 'local-video-view' })
    localVideoStarted.value = true
    mutedVideo.value = false
  } else {
    await trtc.value.stopLocalVideo()
    localVideoStarted.value = false
    mutedVideo.value = true
  }
}

async function hangup() {
  await cleanup(true)
  uni.navigateBack()
}
```
